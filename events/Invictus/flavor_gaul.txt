flavor_gaul.6 = {                            # Lugian League - START
    type = country_event
    title = flavor_gaul.6.t
    desc = flavor_gaul.6.desc
    picture = throneroom
    left_portrait = current_ruler
    right_portrait = scope:league_1.current_ruler
    right_portrait = scope:league_2.current_ruler
    right_portrait = scope:league_3.current_ruler

    trigger = {
        has_land = yes
    }

    immediate = {
        ordered_country = {
            limit = {
                primary_culture = vandal
            }
            max = 3
            check_range_bounds = no
            order_by = country_population
            if = {
                limit = {
                    NOT = { exists = scope:league_1 }
                }
                save_scope_as = league_1
            }
            else_if = {
                limit = {
                    exists = scope:league_1
                    NOT = { THIS = scope:league_1 }
                }
                save_scope_as = league_2
            }
            else_if = {
                limit = {
                    exists = scope:league_1
                    exists = scope:league_2
                    NOR = {
                        THIS = scope:league_1
                        THIS = scope:league_2
                    }
                }
                save_scope_as = league_3
            }
            add_to_list = bohemian_league_list
        }
        add_political_influence = -100
    }

    option = {
        name = flavor_gaul.6.a
        custom_tooltip = bohemia_lugian_league_reward_tt
    }

    after = {
        every_in_list = {
            list = bohemian_league_list
            trigger_event = flavor_gaul.7
        }
    }
}

flavor_gaul.8 = {                            # Lugian League - END
    type = country_event
    title = flavor_gaul.8.t
    desc = {
        desc = flavor_gaul.8.desc
        first_valid = {
            triggered_desc = {
                trigger = {
                    any_in_list = {
                        list = bohemian_league_list
                        has_variable = bohemian_subject_league_flag
                    }
                }
                desc = flavor_gaul.8.desc_1 # someone said yes
            }
            triggered_desc = {
                trigger = { always = yes }
                desc = flavor_gaul.8.desc_2 # noone said yes
            }
        }
    }
    picture = greek_political_discussion
    left_portrait = current_ruler
    right_portrait = scope:lugian_league_head_scope.current_ruler
    goto_location = scope:lugian_league_head_scope.capital_scope

    trigger = {
        has_land = yes
    }

    immediate = {
        if = {
            limit = {
                any_in_list = {
                    list = bohemian_league_list
                    has_variable = bohemian_subject_league_flag
                }
            }
            if = {
                limit = {
                    any_in_list = {
                        list = bohemian_league_list
                        has_variable = bohemian_subject_league_flag
                        is_ai = no
                    }
                }
                random_in_list = {
                    list = bohemian_league_list
                    limit = {
                        has_variable = bohemian_subject_league_flag
                        is_ai = no
                    }
                    save_scope_as = lugian_league_head_scope
                }
            }
            else = {
                ordered_in_list = {
                    list = bohemian_league_list
                    limit = { has_variable = bohemian_subject_league_flag }
                    order_by = country_population
                    save_scope_as = lugian_league_head_scope
                }
            }
            set_variable = {
                name = bohemian_league_counter_var
                value = 0
            }
            every_in_list = {
                list = bohemian_league_list
                limit = {
                    has_variable = bohemian_subject_league_flag
                    NOT = { THIS = scope:lugian_league_head_scope }
                }
                ROOT = {
                    change_variable = {
                        name = bohemian_league_counter_var
                        add = 1
                    }
                }
                remove_variable = bohemian_subject_league_flag
                every_owned_province = { set_conquered_by = scope:lugian_league_head_scope }
            }
            hidden_effect = {
                scope:lugian_league_head_scope = {
                    change_country_name = LGI
					change_country_adjective = "LGI_ADJ"
					change_country_color =  "lugia_color"
					change_country_tag = LGI
					change_country_flag = LGI
                    if = {
                        limit = { is_subject = no }
                        ROOT = {
                            make_subject = {
                                target = scope:lugian_league_head_scope
                                type = client_state
                            }
                        }
                    }
                }
            }
        }
        else = { custom_tooltip = flavor_gaul.8.tt1 }
    }

    option = {
        name = flavor_gaul.8.a
        custom_tooltip = flavor_gaul.8.at
    }

    after = {
        remove_variable = bohemian_league_counter_var
    }
}
