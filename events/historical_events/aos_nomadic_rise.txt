namespace = aos_nomadic_rise
aos_nomadic_rise.1 = { #游牧崛起
	
	type = country_event
	title = "aos_nomadic_rise.1.t"
	desc = "aos_nomadic_rise.1.desc"
	picture = xiongnu_cavalry
	goto_location = root
	left_portrait = current_ruler
	fire_only_once = yes

	trigger = {
		OR = {
			country_culture_group = di
			country_culture_group = hu
		}
		NOT = {
			has_country_modifier = aos_nomadic_rise_buff
			has_country_modifier = nomadic_empire
			has_country_modifier = aos_nomadic_cavalry
		}
		num_of_cities >= 28
	}
	
	option = {
		name = aos_nomadic_rise.1.a
		ai_chance = {
			modifier = {
				factor = 100
			}
		}
		add_country_modifier = {
			name = aos_nomadic_rise_buff
			duration = 18250
		}
		add_country_modifier = {
			name = aos_nomadic_cavalry
			duration = -1
		}
		region:mobei_region = { formable_region_province_effect = yes }
		region:shuofang_region = { formable_region_province_effect = yes }
		region:xiyu_region = { formable_region_province_effect = yes }
		region:yong_region = { formable_region_province_effect = yes }
	}
}
