namespace = clean_up

clean_up.1 = { #mint
	type = country_event
	hidden = yes

	trigger = {
		mint_current_amount > mint_max_amount
	}

	immediate = {
		ordered_owned_province = {
			limit = {
				has_building = mint_building
			}
			order_by = {
				value = 0
				add = total_population
				multiply = -1
			}
			remove_building_level = mint_building
		}
	}
}
