me_seleukid_02_syria.16 = { # Finding Greek Settlers
    type = province_event
    hidden = yes

    trigger = {
        has_city_status = yes
        total_population >= 20
    }

    immediate = {
        scope:seleukid_nation = {
            random_owned_province = {
                limit = {
                    has_city_status = yes
                    OR = {
                        is_in_area = syria_area
                        is_in_area = syria_cyrrestice_area
                    }
                }
                weight = {
                    modifier = {
                        factor = 10
                        is_port = yes
                    }
                    modifier = {
                        factor = 5
                        is_capital = yes
                    }
                    modifier = {
                        factor = 5
                        is_state_capital = yes
                    }
                    modifier = {
                        factor = 3
                        total_population < population_cap
                    }
                    modifier = {
                        factor = 0.005
                        total_population > population_cap
                    }
                }
                save_scope_as = seleukid_city
            }
        }
        while = {
            count = { 5 15 }
            if = {
                limit = {
                    total_population >= 20
                    any_pops_in_province = {
                        pop_culture_group = hellenic
                        NOR = {
                            pop_type = slaves
                            pop_type = tribesmen
                        }
                    }
                }
                random_pops_in_province = {
                    limit = {
                        pop_culture_group = hellenic
                        NOR = {
                            pop_type = slaves
                            pop_type = tribesmen
                        }
                    }
                    move_pop = scope:seleukid_city
                }
                if = {
                    limit = {
                        has_variable = seleukid_pop_move_variable
                    }
                    change_variable = {
                        name = seleukid_pop_move_variable
                        add = 1
                    }
                }
                else = {
                    set_variable = {
                        name = seleukid_pop_move_variable
                        value = 1
                    }
                }
                scope:seleukid_city = {
                    if = {
                        limit = {
                            has_variable = seleukid_pop_move_variable
                        }
                        change_variable = {
                            name = seleukid_pop_move_variable
                            add = 1
                        }
                    }
                    else = {
                        set_variable = {
                            name = seleukid_pop_move_variable
                            value = 1
                        }
                    }
                }
            }
        }
        if = {
            limit = {
                has_variable = seleukid_chosen_city
            }
            remove_variable = seleukid_chosen_city
        }
    }
}
