#培养继承人
aos_heir_events.1 = {
	type = country_event
	title = "aos_heir_events.1.t"
	desc = "aos_heir_events.1.desc"
	left_portrait = current_ruler
	right_portrait = current_heir

	trigger = {
		is_ai = no
		is_monarchy = yes
		current_ruler = {
			NOT = {
				has_trait = cruel
				has_trait = cold
			}
			finesse > 5
		}
		current_heir = {
			is_child_of = root.current_ruler
			is_primary_heir = yes
			prisoner = no
			age > 5
			age < 16
		}
	}
	
	immediate = {
		current_heir = {
			save_scope_as = rulers_heir_character
		}
		current_ruler = {
			save_scope_as = heirs_ruler
		}
	}
	
	option = {
		name = "aos_heir_events.1.a"
		trigger_event = {
			id = aos_heir_events.2
		}
	}

	option = {
		name = "aos_heir_events.1.b"
		scope:heirs_ruler = {
			add_health = 0.1
		}
	}
}
