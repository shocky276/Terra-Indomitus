namespace = character_study_events

#开始学习
character_study_events.1 = {
	type = country_event
	title = "character_study_events.1.t"
	desc = "character_study_events.1.desc"
	left_portrait = current_ruler

	trigger = {
		is_ai = no
	}
	
	immediate = {
	current_ruler = {
		add_trait = studying_trait
		}
	}
	
	option = {
	name = "character_study_events.1.a"
			random_list = {
			90 = {
				trigger_event = {
					id = character_study_events.2
					days = { 365 1825 }
				}
			}
			10 = {
				trigger_event = {
					id = character_study_events.3
					days = { 180 500 }
				}
			}
		}
	}
}

#学有所成
character_study_events.2 = {
	type = country_event
	title = "character_study_events.2.t"
	desc = "character_study_events.2.desc"
	left_portrait = current_ruler

	trigger = {
		current_ruler = {
			has_trait = studying_trait
		}
	}

	immediate = {
		current_ruler = {
			remove_trait = studying_trait
		}
	}

	option = {
		name = "character_study_events.2.a" #在军略方面取得进步
		trigger = {
			current_ruler = {
				martial < 10
			}
		}
			current_ruler = {
				add_martial = 1
			}
	}

	option = {
		name = "character_study_events.2.b" #在智略方面取得进步
		trigger = {
			current_ruler = {
				finesse < 12
			}
		}
			current_ruler = {
				add_finesse = 1
			}
	}

	option = {
		name = "character_study_events.2.c" #在魅力方面取得进步
		trigger = {
			current_ruler = {
				charisma < 10
			}
		}
			current_ruler = {
				add_charisma = 1
			}
	}

	option = {
		name = "character_study_events.2.d" #在热忱方面取得进步
		trigger = {
			current_ruler = {
				zeal < 10
			}
		}
			current_ruler = {
				add_zeal = 1
			}
	}

	option = {
		name = "character_study_events.2.e"
		trigger = {
			current_ruler = {
				martial > 10
				finesse > 12
				charisma > 10
				zeal > 10
			}
		}
	}
}
