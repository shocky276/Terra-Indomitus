namespace = dynasty_end_continuation
dynasty_end_continuation.1 = {
	type = country_event
	title = "dynasty_end_continuation.1.t"
	desc = "dynasty_end_continuation.1.desc"
	picture = china_political
	goto_location = root
	left_portrait = current_ruler

	trigger = {
	has_variable = china_is_unified
	}
	
	option = {
		name = dynasty_end_continuation.1.a
		add_stability = -12
		add_treasury = -300
		add_political_influence = -30
		add_country_modifier = {
		name = gaigetudi_buff
		duration = 3650
		}
		trigger_event = {
			id = dynasty_end_continuation.2 #判断统治者能力，改革成功或失败
			days = {180 900}
		}
	}

	option = {
		name = dynasty_end_continuation.1.b
		add_country_modifier = {
			name = tudijianbing1_buff
		}
		trigger_event = {
			id = dynasty_end_continuation.3 #30-50年后土地兼并2级
			days = {10950 18250}
		}
	}
}

dynasty_end_continuation.2 = {
	type = country_event
	title = "dynasty_end_continuation.2.t"
	desc = "dynasty_end_continuation.2.desc"
	picture = china_political
	goto_location = root
	left_portrait = current_ruler

	trigger = {
	has_variable = china_is_unified
	has_country_modifier = gaigetudi_buff
	}
	
	option = {
		name = dynasty_end_continuation.2.a
		trigger = {
			current_ruler = {
				finesse < 8
			}
		}
		trigger_event = {
			id = dynasty_end_continuation.3 #改革失败，2级土地兼并
			days = {10950 18250}
		}
		remove_country_modifier = gaigetudi_buff
		add_country_modifier = {
			name = gaigeshibai_buff
			duration = 3650
		}
		add_country_modifier = {
			name = tudijianbing1_buff
		}
		add_stability = -10
		add_political_influence = -10
	}

	option = {
		name = dynasty_end_continuation.2.b
		trigger = {
			current_ruler = {
				finesse > 8
			}
		}
		trigger_event = {
			id = dynasty_end_continuation.1 #改革成功，30-50年后回到第一个事件抑制豪强崛起
			days = {10950 18250}
		}
	}
}
