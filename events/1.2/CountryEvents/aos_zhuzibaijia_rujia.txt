rujia_events.1 = { #止息兵戈
	type = country_event
	title = "rujia_events.1.t"
	desc = "rujia_events.1.desc"
	picture = china_political
	left_portrait = current_ruler
	right_portrait = scope:rujia_minister_target
	
	trigger = {
		NOT = {
			has_country_modifier = yanlushouzu_buff
			has_variable = ruzhejiesheng
		}
		war = yes
		has_war_exhaustion > 16
		country_culture_group = chinese
		any_character = {
			prisoner = no
			is_adult = yes
			has_trait = rujia_trait
			is_ruler = no
			is_female = no
		}
	}

	immediate = {
		random_character = {
			limit = {
				prisoner = no
				is_adult = yes
				is_female = no
				is_ruler = no
				has_trait = rujia_trait
			}
			save_scope_as = rujia_minister_target
		}
	}

	option = { #短期内承诺停战
		name = "rujia_events.1.a"
		trigger = {
			current_ruler = {
				NOT = {
					has_trait = abrasive
					has_trait = cruel
				}
			}
		}
		scope:rujia_minister_target = {
			add_loyalty = aos_loyalty_adopt_suggestion
		}
		trigger_event = {
			id = rujia_events.2
			days = 180
		}
	}

	option = { #一年内承诺停战
		name = "rujia_events.1.b"
		trigger = {
			current_ruler = {
				NOT = {
					has_trait = abrasive
					has_trait = cruel
				}
			}
		}
		scope:rujia_minister_target = {
			add_loyalty = aos_loyalty_adopt_suggestion_small
		}
		trigger_event = {
			id = rujia_events.2
			days = 365
		}
	}

	option = { #能言善辩的君主否决但不减忠诚度
		name = "rujia_events.1.c"
		trigger = {
			current_ruler = {
				has_trait = silver_tongued
				has_trait = orator
			}
		}
	}

	option = { #怒斥反驳
		name = "rujia_events.1.d"
		custom_tooltip = abrasive_tooltip
		scope:rujia_minister_target = {
			add_loyalty = aos_loyalty_be_insulted
		}
		add_country_modifier = {
			name = yanlushouzu_buff
			duration = 2920
		}
	}

	option = { #关起来
		name = "rujia_events.1.e"
		custom_tooltip = abrasive_tooltip
		trigger = {
			current_ruler = {
				has_trait = cruel
			}
		}
		scope:rujia_minister_target = {
			remove_all_positions = yes
			add_loyalty = aos_loyalty_be_insulted
			root = {
				imprison = {
					target = scope:rujia_minister_target
				}
			}
		}
		current_ruler = {
			add_popularity = -40
		}
		add_country_modifier = {
			name = yanlushouzu_buff
			duration = 3650
		}
	}

	option = { #君主是法家或以法家治国
		name = "rujia_events.1.f"
		custom_tooltip = fajia_trait_tooltip
		trigger = {
			OR = {

rujia_events.2 = { #止息兵戈判定是否还在战争状态
	type = country_event
	title = "rujia_events.2.t"
	desc = {
		first_valid = {
			triggered_desc = { #停战了
				trigger = {
					war = no
					local_var:rujia_armistice = 1
				}
				desc = "rujia_events.2.armistice"
			}
			triggered_desc = { #还在打
				trigger = {
					war = yes
					local_var:rujia_armistice = 0
				}
				desc = "rujia_events.2.at_war"
			}
		}
	}

	picture = china_political
	left_portrait = current_ruler
	right_portrait = scope:rujia_minister_target

	immediate = {
		if = {
			limit = {
				war = no
			}
			set_local_variable = {
				name = rujia_armistice
				value = 1
			}
		}
		else = {
			set_local_variable = {
				name = rujia_armistice
				value = 0
			}
		}
	}

	option = {
		name = "rujia_events.2.armistice.o"
		trigger = {
			has_local_variable = rujia_armistice
			local_var:rujia_armistice = 1
		}
		scope:rujia_minister_target = {
			add_loyalty = aos_loyalty_pay_promise_tiny
		}
	}

	option = {
		name = "rujia_events.2.at_war.o"
		trigger = {
			has_local_variable = rujia_armistice
			local_var:rujia_armistice = 0
		}
		scope:rujia_minister_target = {
			add_loyalty = aos_loyalty_be_deceived
		}
	}
}
