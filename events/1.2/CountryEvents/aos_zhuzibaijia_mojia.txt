namespace = aos_mojia_events

aos_mojia_events.1 = { #制造连弩车
	type = country_event
	title = "aos_mojia_events.1.t"
	desc = "aos_mojia_events.1.desc"
	picture = china_political
	left_portrait = current_ruler
	right_portrait = scope:mojia_minister_target
	
	trigger = {
		NOT = {
			has_country_modifier = yanlushouzu_buff
			exists = treasure:treasure_liannuche
		}
		is_ai = no
		country_culture_group = chinese
		any_character = {
			is_adult = yes
			prisoner = no
			has_trait = mojia_trait
			is_ruler = no
			is_female = no
		}
	}

	immediate = {
		random_character = {
			limit = {
				is_adult = yes
				prisoner = no
				has_trait = mojia_trait
				is_ruler = no
				is_female = no
			}
			save_scope_as = mojia_minister_target
		}
	}

	option = { #大批装备
		name = "aos_mojia_events.1.a"
		trigger = {
			current_ruler = {
				NOT = {
					has_trait = abrasive
					has_trait = cruel
				}
				OR = {
					finesse > 6
					has_trait = mojia_trait
				}
			}
		}
		create_country_treasure = {
			key = treasure_liannuche
			icon = treasure_liannuche

			modifier = {
				siege_ability = 0.2
			}
		}
		scope:mojia_minister_target = {
			add_loyalty = aos_loyalty_adopt_suggestion
		}
		add_political_influence = { -10 -20 }
		add_treasury = { -1300 -1800 }
		add_country_modifier = {
			name = treasure_liannuche_buff
			duration = -1
		}
	}

	option = { #婉拒
		name = "aos_mojia_events.1.b"
		trigger = {
			current_ruler = {
				NOT = {
					has_trait = abrasive
					has_trait = cruel
				}
			}
		}
		scope:mojia_minister_target = {
			add_loyalty = aos_loyalty_suggestion_be_perfunctory
		}
	}

	option = { #怒骂
		name = "aos_mojia_events.1.c"
		custom_tooltip = abrasive_tooltip
		trigger = {
			current_ruler = {
				OR = {
					has_trait = abrasive
					has_trait = cruel
				}
			}
		}
		scope:mojia_minister_target = {
			add_loyalty = aos_loyalty_be_insulted
		}
	}

	option = { #关起来
		name = "aos_mojia_events.1.d"
		custom_tooltip = abrasive_tooltip
		trigger = {
			current_ruler = {
				has_trait = cruel
			}
		}
		scope:mojia_minister_target = {
			remove_all_positions = yes
			add_loyalty = aos_loyalty_be_insulted
			root = {
				imprison = {
					target = scope:mojia_minister_target
				}
			}
		}
		current_ruler = {
			add_popularity = -40
		}
		add_country_modifier = {
			name = yanlushouzu_buff
			duration = 3650
		}
	}
}
