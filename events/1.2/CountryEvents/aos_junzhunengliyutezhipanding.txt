namespace = real_mode_events

real_mode_events.1 = {
	type = country_event
	title = real_mode_events.1.t
	desc = real_mode_events.1.desc
	left_portrait = current_ruler

	trigger = {
		is_ai = no
	}

	option = { #启用
		name = real_mode_events.1.a
	}

	option = { #不启用
		name = off_real_mode_events.1.b
		set_variable = off_real_mode
	}
}

namespace = give_zhijunbushan_buff

give_zhijunbushan_buff.1 = { #不擅治军
	type = country_event
	title = give_zhijunbushan_buff.1.t
	desc = give_zhijunbushan_buff.1.desc
	left_portrait = current_ruler
	
	trigger = {
		NOT = {
			has_country_modifier = zhijunbushan_buff
			has_variable = regent_variable
			has_variable = off_real_mode
		}
		OR = {
			government = dictatorship
			government = despotic_monarchy
			government = stratocratic_monarchy
		}
		current_ruler = {
			martial <= 2
		}
	}
	
	immediate = { 
		add_country_modifier = {
			name = zhijunbushan_buff
			duration = -1
		}
	}
	
	option = {
		name = give_zhijunbushan_buff.1.a
	}
}
