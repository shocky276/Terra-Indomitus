zhuhouguo_events.1 = {
	type = country_event
	title = "zhuhouguo_events.1.t"
	desc = "zhuhouguo_events.1.desc"
	picture = china_political

	trigger = {
		country_culture_group = chinese
		is_subject = yes
		any_neighbour_country = {
			this = overlord
			country_culture_group = chinese
			num_of_cities > 30
			NOT = {
				has_variable = jiaqiangtianwei
			}
		}
	}

	immediate = {
		root = {
			save_scope_as = zhuhou_country
		}
		overlord = {
			save_scope_as = zhuhou_overlord
		}
		scope:zhuhou_overlord = {
			random_owned_province = {
				limit = {
					any_neighbor_province = {
						owner = scope:zhuhou_country
					}
				}
				save_scope_as = zhuhou_target_province
			}
		}
	}

	option = {
		name = zhuhouguo_events.1.a
		ai_chance = {
			modifier = {
				factor = 100
			}
		}
		scope:zhuhou_overlord = {
			trigger_event = {
				id = zhuhouguo_events.2
			}
		}
	}

	option = {
		name = zhuhouguo_events.1.b
		ai_chance = {
			modifier = {
				factor = 0
			}
		}
	}
}
