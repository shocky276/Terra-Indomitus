	hidden = yes

		}

	}
}

namespace = xianzhikuozhang
xianzhikuozhang.1 = { #领土超过900块debuff
	
	type = country_event
	title = "xianzhikuozhang.1.t"
	desc = "xianzhikuozhang.1.desc"
	picture = china_political
	goto_location = root
	left_portrait = current_ruler

	trigger = {
	NOT = {
	has_country_modifier = 700lingtu_debuff
	}
	num_of_cities >= 900
	}
	
	option = {
		name = xianzhikuozhang.1.a
		ai_chance = {
			modifier = {
				factor = 100
			}
		}
		add_country_modifier = {
		name = 700lingtu_debuff
		duration = -1
		}
	}
}
