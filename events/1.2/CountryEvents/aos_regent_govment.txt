namespace = aos_regent_gov
aos_regent_gov.1 = { #进入监国状态
	
	type = country_event
	title = "aos_regent_gov.1.t"
	desc = "aos_regent_gov.1.desc"
	left_portrait = current_ruler

	trigger = {
		NOT = {
			has_variable = regent_variable
		}
		current_ruler = {
			is_adult = no
		}
	}

	immediate = {
		set_variable = regent_variable
	}
	
	option = {
		name = aos_regent_gov.1.a
		add_country_modifier = {
			name = regent_buff
			duration = -1
		}
	}
}

namespace = aos_remove_regent_gov
aos_remove_regent_gov.1 = { #结束监国状态
	
	type = country_event
	title = "aos_remove_regent_gov.1.t"
	desc = "aos_remove_regent_gov.1.desc"
	left_portrait = current_ruler

	trigger = {
		has_variable = regent_variable
		current_ruler = {
			is_adult = yes
		}
	}

	immediate = {
		remove_variable = regent_variable
	}
	
	option = {
		name = aos_remove_regent_gov.1.a
		remove_country_modifier = regent_buff
	}
}
