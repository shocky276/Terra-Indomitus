diadochi_wargoal = {
	type = superiority
	#uses_civil_war_conquest = yes
	allow = { #Will this show up in the declare war window.
		is_subject = no
		OR = { 
			tag = EGY
			tag = MAC
			tag = TRE
			tag = SEL
			tag = PRY
			has_variable = pleistarchus_satrapy_flag
		}
		NOT = { has_variable = epigoni } #No longer usable if you had the Epigoni event
		has_variable = war_of_the_diadochs #Turns on after the intro event which explains what it does.
		scope:defender = {
			OR = {
				tag = EGY
				tag = MAC
				tag = TRE
				tag = SEL
				tag = PRY
				AND = {
					tag = CAP
					primary_culture = macedonian
				}
				has_variable = pleistarchus_satrapy_flag
				has_variable = asia_satrapy
				has_variable = galatia_satrapy
			}
		}
	}
	
	attacker = {
		conquer_cost = -0.85
	}
	defender = {
		conquer_cost = -0.85
	}
	ticking_war_score = 0.5

}

	type = take_province
	allow = {
		is_subject = no
		NAND = {
			is_ai = yes
			OR = {
				AND = {
					scope:defender = {government = tribal_chiefdom}
					is_tribal = no
				}
				AND = {
					is_tribal = yes
					NOR = {
						primary_culture = xiongnu
						AND = {
							country_culture_group = numidian
							current_date>530.1.1
						}
					}
				}
				AND = {
					tag = MRY
					OR = {
						current_date>521.1.1 #史实阿育王去世年份

							NOR = {
								country_culture_group = pracyan
								country_culture_group = indian
								country_culture_group = aryan
							}
						}
					}
				}
				AND = {
					tag = CAR
					OR = {
						AND = {
							scope:defender = {
								country_culture_group = numidian
								NOT = {primary_culture = massylian}
							}
							current_date<600.1.1
						}
						AND = {
							scope:defender = {primary_culture = massylian}
							current_date<580.1.1
						}
					}
				}
				AND = {
					OR = {
						tag = ARM
						tag = MAP
					}
					current_date<600.1.1
				}
				AND = {
					OR = { 
						tag = EGY
						tag = MAC
						tag = TRE
						tag = SEL
						tag = PRY
					}
					scope:defender = {
						NOR = {
						tag = EGY
						tag = MAC
						tag = TRE
						tag = SEL
						tag = PRY
						}
					}
					NOT = { has_variable = epigoni }
				}
				AND = {
					tag = SEL
					region:palestine_region = {
						any_region_province = {
							owner = c:SEL
						}
					}
					NOT = {
						region:lower_egypt_region = {
							any_region_province = {
								owner = c:SEL
							}
						}
					}
					scope:defender = {
						tag = EGY
					}
				}
				AND = {
					tag = SEL
					scope:defender = {country_culture_group = armenian}
					current_date < 540.1.1
				}
				AND = {
					OR = {
						tag = SEL
						tag = ARM
					}
					scope:defender = {country_culture_group = persia}
					current_date<600.1.1
				}
				AND = {
					NOT = {country_culture_group = caucasian}
					scope:defender = {country_culture_group = caucasian}
					current_date<600.1.1
				}
				AND = {
					NOT = {country_culture_group = albanian}
					scope:defender = {country_culture_group = albanian}
					current_date<550.1.1
				}
				AND = {
					tag = IBR
					current_date<550.1.1
				}
				AND = {
					NOT = {country_culture_group = scythia}

				}
