paradisial_garden = {

	on_other_nation = no
	on_own_nation = yes
	message = no
	
	sound = "event:/SFX/UI/Character/Unique/sfx_ui_character_give_free_hands"
	
	potential_trigger = {
		hidden:scope:actor = {
			invention = paradisial_gardens_inv
			NOT = { has_variable = gardens_cooldown }
			OR = {
				country_culture_group = persia
				any_country_culture = {
					is_integrated = yes
					OR = {
						is_culture = elamite
						is_culture = utian
						is_culture = pasargadi
						is_culture = sagartian
						is_culture = uxian
						is_culture = amardian
						is_culture = sarangian
						is_culture = pactyan
						is_culture = cossian
						is_culture = cadusian
						is_culture = parthian
						is_culture = parecanian
						is_culture = carmanian
						is_culture = median
						is_culture = gedrosian
						is_culture = hyrcanian
						is_culture = sattagydian
					}
					country_culture_pop_count >= 350
				}
			}
		}
		scope:target = {
			employer = scope:actor
			is_alive = yes
			NOT = { has_variable = olympic_attendee }
		}
	}

	allowed_trigger = {
		hidden:scope:actor = {
			treasury >= 100
			political_influence >= 10
		}
		hidden:scope:target = {
			NOT = {
				has_character_modifier = foreign_citizen
			}
			prisoner = no
			is_mercenary = no
			NOT = { has_trait = gluttonous }
			wealth >= 150
			any_holdings = {
				NOT = {
					has_province_modifier = paradisial_gardens_modifier
				}
			}
		}
	}
	
	effect = {
		scope:actor = {
			add_treasury = -100
			add_political_influence = -10
			set_variable = {
				name = gardens_cooldown
				duration = 12775 #35 yrs
			}
			scope:target = {
				add_gold = -100
				ordered_holdings = {
					order_by = total_population
					limit = {
						NOT = {
							has_province_modifier = paradisial_gardens_modifier
						}
					}
					add_province_modifier = {
						name = paradisial_gardens_modifier
						duration = 12775 #35 Years
					}
				}
			}
		}	
	}
	ai_will_do = {
		base = 1
		modifier = {
			add = 10
			scope:actor = {
				treasury >= 1000
			}
		}
		modifier = {
			add = 20
			scope:actor = {
				treasury >= 2000
			}
		}
		modifier = {
			add = -10
			scope:actor = {
				treasury <= 500
			}
		}
	}
	
}
