taijian_assassinate = {

	on_other_nation = no
	on_own_nation = yes
	message = no
	
	potential_trigger = {
		any_country = {
			is_ai = no
			current_ruler = {
				has_variable = huanguanzhangquan_ruler
			}
		}
		hidden:scope:target = {
			is_alive = yes
			is_ruler = no
		}
	}
	
	allowed_trigger = {
		hidden:scope:target = {
			is_ruler = no
			is_mercenary = no
		}
		hidden:scope:actor = {
			can_pay_price = assassinate_attempt
		}
		scope:target = {
			NOT = {
				is_friend = scope:actor.current_ruler
			}
		}
		custom_tooltip = {
			text = employer_is_target_employer_tt
			scope:actor.current_ruler = {
				any_friend = {
					employer = scope:target.employer
					NOR = {
						is_ruler = yes
						is_close_relative = scope:target
						is_spouse_of = scope:target
					}
				}
			}
		}
		custom_tooltip = {
			text = only_one_assassination
			scope:actor = {
				NOT = {
					has_variable = ongoing_assassination
				}
			}
		}
	}
	
	
	effect = {
		scope:actor = {
			pay_price = assassinate_attempt
			set_variable = {
				name = ongoing_assassination
				days = 360
			}
		}
		custom_tooltip = assassinate_attempt_effect_tooltip
		hidden_effect = {
			scope:actor.current_ruler = {
				random_friend = {
					limit = {
						employer = scope:target.employer
					}
					save_scope_as = foreign_agent
				}
			}
			scope:actor = {
				trigger_event = {
					on_action = assassinate_setup
				}
			}
		}
	}
	
}	

taijian_xiazhaoyu = {

	on_other_nation = no
	on_own_nation = yes
	message = no
	sound = "event:/SFX/UI/Character/Unique/sfx_ui_character_accuse_crime"
	
	potential_trigger = {
		any_country = {
			is_ai = no
			current_ruler = {
				has_variable = huanguanzhangquan_ruler
			}
		}
		hidden:scope:target = {
			is_alive = yes
			is_ruler = no
		}
	}
	
	allowed_trigger = {
		hidden:scope:target = {
			is_ruler = no
			is_mercenary = no
		}
		hidden:scope:actor = {
			can_pay_price = price_bring_to_trial
		}
		scope:target = {
			custom_tooltip = {
				text = office_trial_success_chance
				get_trial_success_chance_character >= 1
			}
		}
		scope:target = {
			custom_tooltip = {
				text = "family_favor_imprison"
				NOT = { 
					has_variable = cannot_imprison_family_favor
				}
			}
			NOT = {
				has_triggered_character_modifier = currently_on_trial
			}
			trigger_if = {
				limit = {
					has_variable = recent_trial_cooldown_var
				}
				custom_tooltip = {
					text = "recent_trial_cannot_tt"
					NOT = {
						has_variable = recent_trial_cooldown_var
					}
				}
			}
		}
	}

	
	effect = {
		if = {
			limit = { scope:target = { get_trial_success_chance_character >= 1 } }
			custom_tooltip = office_trial_success_chance
		}
		if = {
			limit = { scope:target.power_base > 0 }
			custom_tooltip = chances_reduced_by
		}
		if = {
			limit = { scope:target.corruption > 0 }
			custom_tooltip = chances_increased_by
		}
		if = {
			limit = {
				NOT = { scope:target = { has_culture = employer.culture } }
				scope:actor = {
					any_country_culture = {
						has_country_culture_modifier = trial_right
					}
				}
			}
			custom_tooltip = easier_to_prosecute_tt
		}
		custom_tooltip = taijian_xiazhaoyu_success_can_improve
		hidden_effect = {
			scope:target = {
				add_triggered_character_modifier = {
					name = currently_on_trial 
					duration = 730
				}
				add_loyalty = currently_on_trial
				set_variable = {
					name = recent_trial_cooldown_var
					days = 1825
				}
			}
		}
		scope:actor = {
			pay_price = price_bring_to_trial
			trigger_event = {
				id = taijian_xiazhaoyu_event.1
			}
		}
		scope:target = {
			add_loyalty = brought_to_trial_l
		}
		custom_tooltip = trial_consequences
		if = {
			limit = { 
				exists = scope:sen_obj_imp_hea_gen_target
				scope:target = scope:sen_obj_imp_hea_gen_target
			}
			scope:actor = { set_variable = trail_agenda_flag }
		}
	}
	
}
