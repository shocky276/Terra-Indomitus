castration_character = {

	on_other_nation = no
	on_own_nation = yes
	needs_senate_approval = yes
	
	potential_trigger = {
		hidden:scope:target = {
			employer = scope:actor
			NOT = { employer = { tag = REB } }
			NOT = { employer = { tag = BAR } }
			prisoner_valid_for_country_events_trigger = yes
			NOT = {
				has_character_modifier = disloyal_successor_cmod
			}
			prisoner = yes
		}
	}
	
	allowed_trigger = {
		hidden:scope:actor = {
			can_pay_price = price_execute
		}
		hidden:scope:target = {
			NOT = {
				has_trait = castrated
			}
			is_male = yes
			prisoner = yes
			is_co_ruler = no
		}
	}

	effect = {
		scope:actor = {
			pay_price = price_execute
		}
		scope:target = {
			add_trait = castrated
			add_rival = scope:actor.current_ruler
			add_health = -25
		}
	}
}
