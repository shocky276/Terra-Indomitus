		limit = {
			exists = owner
			owner = {
				OR = {

					tag = EGY
					tag = TRE
					tag = PRY

					AND = {
						is_subject = yes
						overlord = {
							NOT = { THIS = ROOT }
							is_diadochi_trigger = yes
						}
					}
				}
			}
			NOT = { 
				owned_or_subject_owned = ROOT
			}
		}
		add_claim = ROOT
	}
}
